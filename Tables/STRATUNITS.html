CREATE TABLE STRATUNITS (
    UNITID             UNITID DEFAULT 0 NOT NULL,
    COUNTRYID          COUNTRYID DEFAULT 'ZAF' NOT NULL,
    UNITNAME           UNITNAME DEFAULT 'Not defined' NOT NULL,
    RANKID             RANKID DEFAULT 'FM' NOT NULL,
    PARENTID           UNITID DEFAULT 0 NOT NULL,
    APPROVALID         APPROVALID DEFAULT 'IF' NOT NULL,
    AGEYEARS           AGEYEARS DEFAULT 0 NOT NULL,
    MINAGE             AGEMIN DEFAULT 0.0 NOT NULL,
    MAXAGE             AGEMAX DEFAULT 4570.0 NOT NULL,
    AGEUNITS           AGEUNITS DEFAULT 'Ma' NOT NULL,
    CONSTRAINTLEVELID  AGECONSTRAINTLEVELID DEFAULT 0 NOT NULL,
    UNITOFFSET_MM      OFFSET_MM DEFAULT 5 NOT NULL,
    DESCRIPTION        UNITDESCRIPTION2,
    PREVIOUSPARENTID   PREVIOUSPARENTID,
    GISID              GISID DEFAULT 0 NOT NULL,
    DUMMY              APPLICATIONID DEFAULT 2,
    LIPID              LIPID DEFAULT 0 NOT NULL,
    AGERANGE           AGES DEFAULT 0.0 NOT NULL,
    NEEDSCHANGE        T_YESNO DEFAULT 'Y' NOT NULL
);
