CREATE TABLE DOMAINS (
    DOMAINID        DOMAINID NOT NULL,
    CONTINENTID     CONTINENTID DEFAULT 'UND' NOT NULL,
    DOMAINNAME      DOMAINNAME DEFAULT 'not defined' NOT NULL,
    SPOTLATITUDE    LATITUDES DEFAULT 90.0 NOT NULL,
    SPOTLONGITUDE   LONGITUDES DEFAULT 0.0 NOT NULL,
    DOMAINPARENTID  DOMAINID DEFAULT 0 NOT NULL,
    DOMAINTYPEID    DOMAINTYPEID DEFAULT 'nd' NOT NULL,
    MAXAGE          AGEMAX DEFAULT 4570.0 NOT NULL,
    MINPLOTAGE      AGEMIN DEFAULT 0 NOT NULL,
    MAXPLOTAGE      AGEMAX DEFAULT 4570.0 NOT NULL,
    NEEDSCHANGE     T_YESNO DEFAULT 'Y' NOT NULL
);


ALTER TABLE DOMAINS ADD CONSTRAINT UNQ1_DOMAINSNAME UNIQUE (DOMAINNAME);
ALTER TABLE DOMAINS ADD CONSTRAINT PK_DOMAINS PRIMARY KEY (DOMAINID);
ALTER TABLE DOMAINS ADD CONSTRAINT FK_DOMAINS_CHNG FOREIGN KEY (NEEDSCHANGE) REFERENCES YESNO (YESNOID) ON UPDATE CASCADE;
ALTER TABLE DOMAINS ADD CONSTRAINT FK_DOMAINS_CONTINENTID FOREIGN KEY (CONTINENTID) REFERENCES CONTINENTS (CONTINENTID) ON DELETE CASCADE ON UPDATE CASCADE
  USING INDEX FK_DOMAINS_CONT;
ALTER TABLE DOMAINS ADD CONSTRAINT FK_DOMAINS_DOMAINTYPEID FOREIGN KEY (DOMAINTYPEID) REFERENCES DOMAINTYPES (DOMAINTYPEID) ON UPDATE CASCADE
  USING INDEX FK_DOMAINS_DOMAINTYPE;
