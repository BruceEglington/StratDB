CREATE TRIGGER RPL$LPRE_DUCTILITIES_D FOR DUCTILITIES
ACTIVE AFTER DELETE POSITION 32766
AS
declare variable primary_key_values varchar(500);
declare variable QuotedValue varchar(200);
declare variable user_login varchar(50);
declare variable repl_operation char(1);
declare variable replicating_node varchar(50);
declare variable primary_key_sync varchar(5000);
declare variable unique_key_sync varchar(5000);
declare variable operation_type char(1);
begin
  replicating_node = rdb$get_context('USER_SESSION', 'REPLICATING_NODE');
  primary_key_values = '';
  primary_key_values = '';
  if (old."DUCTILITYTYPEID" is null) then QuotedValue = '"';
  else select quoted_str from RPL$quote_str(old."DUCTILITYTYPEID") into :QuotedValue;
  primary_key_values = primary_key_values || QuotedValue || ';';

if (inserting) then operation_type = 'I';
else if (updating) then operation_type = 'U';
else operation_type = 'D';
  for select u.login, t.repl_deletes
    from RPL$users u
    join rpl$tables_config t on t.table_name = 'DUCTILITIES' and t.config_name = 'LPRE'
    where (u.login <> :replicating_node or :replicating_node is null)
    and (u.config_name is null or u.config_name = 'LPRE')
    into :user_login, :repl_operation do
  begin
    if (repl_operation = 'Y') then
      insert into RPL$log (code, login, operation_date, table_name, sent_from,
        primary_key_values, procedure_statement, PRIMARY_KEY_SYNC, UNIQUE_KEY_SYNC, OPERATION_TYPE)
      values (gen_id(gen_rpl$log, 1), :user_login, current_timestamp, 'DUCTILITIES', :replicating_node,
        :primary_key_values, null, :PRIMARY_KEY_SYNC, :UNIQUE_KEY_SYNC, :OPERATION_TYPE);
  end
end
