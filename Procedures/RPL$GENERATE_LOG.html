CREATE PROCEDURE RPL$GENERATE_LOG(
    TABLE_NAME VARCHAR(100),
    PRIMARY_KEY_VALUES VARCHAR(500),
    REF_VALUE VARCHAR(50),
    PROCEDURE_STATEMENT VARCHAR(200),
    USER_LOGIN VARCHAR(50),
    PRIMARY_KEY_SYNC VARCHAR(5000),
    UNIQUE_KEY_SYNC VARCHAR(5000))
AS
declare variable replicating_node varchar(50);
begin
  replicating_node = rdb$get_context('USER_SESSION', 'REPLICATING_NODE');
  insert into RPL$log (code, login, operation_date, table_name, sent_from,
    primary_key_values, ref_value, procedure_statement, PRIMARY_KEY_SYNC, UNIQUE_KEY_SYNC)
  values (gen_id(GEN_RPL$LOG, 1), :user_login, 'now', :table_name, :replicating_node,
    :primary_key_values, :ref_value, :procedure_statement, :PRIMARY_KEY_SYNC, :UNIQUE_KEY_SYNC);
end
